#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <stdlib.h>
//                              p        pp
// (input - sent to first pipe) | tr a b | (output - read back in)


int main (void) {
  int p[2];
  int pp[2];
  pipe(p);
  pipe(pp);
  char buf[256];

  if(fork() == 0) {
    if(fork() == 0) {
      close(pp[0]);
      close(pp[1]);
      close(p[0]);
      write(p[1], "staff", strlen("staff"));
    } else {
      close(pp[0]);
      close(p[1]);
      dup2(p[0], STDIN_FILENO);
      dup2(pp[1], STDOUT_FILENO);
      execl("/bin/sh", "sh", "-c", "tr a b", NULL);
    }
  } else {
    close(pp[1]);
    close(p[0]);
    close(p[1]);
    read(pp[0], buf, 32);
    printf("%s\n", buf);
  }
}
